usp_GetStudentAnalytics.sql
-- AnalyticVue · Clarke County SD
CREATE OR ALTER PROCEDURE
  dbo.usp_GetStudentAnalytics
  @DistrictID INT,
  @SchoolYear NVARCHAR(9)
AS BEGIN
  SET NOCOUNT ON;
  -- Indexed covering query
  SELECT s.StudentID,
    s.LastName, s.FirstName,
    e.GradeLevel, e.SchoolID,
    a.AbsenceRate
  FROM main.k12Students s
  JOIN main.k12Enrollment e
    ON s.StudentID = e.StudentID
  WHERE s.DistrictID = @DistrictID
    AND e.SchoolYear = @SchoolYear;
END
 
-- [✓] Execution complete · 0ms
